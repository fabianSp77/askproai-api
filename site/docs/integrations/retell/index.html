    Retell.ai Integration¶

    Overview¶

    Retell.ai provides the AI-powered phone agent that handles incoming calls, conducts conversations in natural language, and extracts appointment booking information. The integration supports multiple languages and custom conversation flows.

    Configuration¶

    Environment Variables¶

    # Retell.ai Configuration
DEFAULT_RETELL_API_KEY=key_xxxxxxxxxxxxxx
DEFAULT_RETELL_AGENT_ID=agent_xxxxxxxxxxxxxx
RETELL_WEBHOOK_SECRET=key_xxxxxxxxxxxxxx
RETELL_BASE_URL=https://api.retellai.com

    Company-Level Configuration¶

    // Configure Retell.ai for a company
$company->update([
    'retell_api_key' => 'key_xxxxxxxxxxxxxx',
    'retell_default_agent_id' => 'agent_xxxxxxxxxxxxxx'
]);

// Configure phone number with agent
$phoneNumber->update([
    'retell_agent_id' => 'agent_xxxxxxxxxxxxxx',
    'retell_agent_version' => 'v2.1'
]);

    Webhook Events¶

    call_started¶

    public function handleCallStarted($payload)
{
    $call = Call::create([
        'retell_call_id' => $payload['call_id'],
        'phone_number' => $payload['from_number'],
        'agent_id' => $payload['agent_id'],
        'status' => 'in_progress',
        'started_at' => now()
    ]);

    // Resolve company/branch from phone number
    $phoneResolver = app(PhoneNumberResolver::class);
    $context = $phoneResolver->resolve($payload['to_number']);

    $call->update([
        'company_id' => $context['company_id'],
        'branch_id' => $context['branch_id']
    ]);
}

    public function handleCallEnded($payload)

    call_analyzed¶

    public function handleCallAnalyzed($payload)
{
    $call = Call::where('retell_call_id', $payload['call_id'])->first();

    $call->update([
        'transcript' => $payload['transcript'],
        'recording_url' => $payload['recording_url'],
        'summary' => $payload['call_summary'],
        'sentiment' => $payload['user_sentiment']
    ]);

    // Extract appointment details if booking was made
    if ($appointmentData = $this->extractAppointmentData($payload)) {
        $this->createAppointmentFromCall($call, $appointmentData);
    }
}

    Webhook Security¶

    class VerifyRetellSignature
{
    public function handle($request, $next)
    {
        $signature = $request->header('x-retell-signature');
        $payload = $request->getContent();
        $secret = config('services.retell.webhook_secret');

        $expected = hash_hmac('sha256', $payload, $secret);

        if (!hash_equals($expected, $signature)) {
            abort(401, 'Invalid webhook signature');
        }

        return $next($request);
    }
}

    Call Processing¶

    Phone Number Resolution¶

    class PhoneNumberResolver
{
    public function resolve(string $phoneNumber): array
    {
        $phone = PhoneNumber::where('number', $phoneNumber)
            ->with(['branch.company'])
            ->firstOrFail();

        return [
            'company_id' => $phone->branch->company_id,
            'branch_id' => $phone->branch_id,
            'agent_id' => $phone->retell_agent_id,
            'event_type_id' => $phone->branch->calcom_event_type_id
        ];
    }
}

    Appointment Extraction¶

    class AppointmentExtractor
{
    public function extract(array $callData): ?array
    {
        // Check if booking function was called
        $functionCalls = $callData['function_calls'] ?? [];

        foreach ($functionCalls as $call) {
            if ($call['name'] === 'book_appointment') {
                return $this->parseBookingData($call['arguments']);
            }
        }

        return null;
    }

    private function parseBookingData(array $arguments): array
    {
        return [
            'customer_name' => $arguments['customer_name'],
            'phone' => $arguments['phone_number'],
            'service' => $arguments['service_type'],
            'date' => Carbon::parse($arguments['preferred_date']),
            'time' => $arguments['preferred_time'] ?? null,
            'notes' => $arguments['notes'] ?? ''
        ];
    }
}

    Troubleshooting¶

    Common Issues¶

    "Agent Not Responding"¶

    1. Check agent ID configuration
    2. Verify API key is valid
    3. Check Retell.ai service status

    "Webhook Not Received"¶

    1. Verify webhook URL in Retell dashboard
    2. Check webhook secret configuration
    3. Review server logs for signature failures

    "Language Not Detected"¶

    1. Ensure language detection is enabled
    2. Check supported language list
    3. Verify voice model supports the language
