// SOFORT-FIX f√ºr Tabellen-Scroll
(function() {
    // Fix alle Tabellen-Container
    document.querySelectorAll('.fi-ta-content').forEach(container => {
        // Entferne alle overflow styles
        container.style.removeProperty('overflow');
        container.style.removeProperty('overflow-x');
        container.style.removeProperty('overflow-y');
        
        // Setze nur overflow-x
        container.style.overflowX = 'auto';
        container.style.overflowY = 'visible';
        container.style.maxWidth = '100%';
        container.style.width = '100%';
        
        // Scrollbar sichtbar machen
        container.style.scrollbarWidth = 'thin';
        container.style.webkitOverflowScrolling = 'touch';
        
        console.log('‚úÖ Fixed container:', container);
    });
    
    // Stelle sicher, dass Tabellen expandieren k√∂nnen
    document.querySelectorAll('.fi-ta-table').forEach(table => {
        table.style.width = 'auto';
        table.style.minWidth = 'max-content';
        table.style.tableLayout = 'auto';
    });
    
    // Force Scrollbar anzeigen
    const style = document.createElement('style');
    style.textContent = `
        .fi-ta-content::-webkit-scrollbar {
            height: 10px !important;
            display: block !important;
        }
        .fi-ta-content::-webkit-scrollbar-track {
            background: #f1f1f1 !important;
        }
        .fi-ta-content::-webkit-scrollbar-thumb {
            background: #888 !important;
            border-radius: 5px !important;
        }
    `;
    document.head.appendChild(style);
    
    console.log('‚úÖ Table scroll fix applied!');
})();
