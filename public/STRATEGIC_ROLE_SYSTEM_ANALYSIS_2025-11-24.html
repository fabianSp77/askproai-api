Architektur Deep Dive

5-Level Security Model

Jede Policy implementiert diese 5 Sicherheitsstufen für Multi-Tenant-Isolation

// Level 1: Platform Admin Override if ($user->hasRole('admin')) { return true; // See ALL appointments (no restrictions) } // Level 2: Company Isolation (CRITICAL for Multi-Tenancy) if ($user->company_id !== $appointment->company_id) { return false; // Prevent cross-company access } // Level 3: Branch Isolation if ($user->hasRole('company_manager') && $user->branch_id) { return $user->branch_id === $appointment->branch_id; } // Level 4: Staff Isolation if ($user->hasRole('company_staff') && $user->staff_id) { return $user->staff_id === $appointment->staff_id; } // Level 5: Company Owners/Admins if ($user->hasAnyRole(['company_owner', 'company_admin'])) { return true; // See all company data } return false; // Default: no access
